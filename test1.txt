#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

int main()
{
 double B[8];
 double C[7];
 double D[8];
 double А[7][8];
 for (int i = 0; i < 7; i++)
    {
        double n;
        for (int j = 0; j < 8; j++)
        {
             cin >> n;
             A[i,j] = n;
        }    
    }
    
for (int i = 0; i < 8; i++)
    {
        double maxVal = A[i,0];
        for (int j = 0; j < 7; j++)
        {
            if (A[i,j] > maxVal)
                maxVal = A[i,j];
        }
        B[i] = maxVal;
        
    }
    
    for (int i = 0; i < 8; i++)
        cout << B[i];
        
for (int i = 0; i < 8; i++)
    {
        double minVal = A[0,i];
        for (int j = 0; j < 7; j++)
        {
            if (A[j,i] < minVal)
                minVal = A[j,i];
        }
        C[i] = minVal;
        
    }
    
      for (int i = 0; i < 8; i++)
        cout << C[i];


for (int i = 0; i < 8; i++)
    {
        double arf = 0;
        for (int j = 0; j < 7; j++)
        {
            arf+=A[i,j];
        }
        
        D[i] = arf/7;
        
    }
    
    for (int i = 0; i < 8; i++)
        cout << D[i];

return 0;
}
